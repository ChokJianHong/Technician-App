sources

    - This is where you get your data from 
    - It will show the API and database, or locally to see all the data
    - It does not care from where the data comes from as long as it has it

    - Example

    // lib/data/sources/remote_book_source.dart
    import 'package:http/http.dart' as http;
    import 'dart:convert';

    class RemoteBookSource {
    final http.Client client;

    RemoteBookSource(this.client);

    Future<Map<String, dynamic>> fetchBookDetails(String bookId) async {
        final response = await client.get(
        Uri.parse('https://bookapi.example.com/books/$bookId'),
        );

        if (response.statusCode == 200) {
        return json.decode(response.body);
        } else {
        throw Exception('Failed to fetch book details');
        }
    }
    }
